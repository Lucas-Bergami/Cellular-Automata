GLC:

S -> Tamanho Estados Grupos Regras

Tamanho -> WIDTH  INT_CONST  HEIGHT  INT_CONST
Estados ->  STATE  { Lista_de_Estados  }
Lista_de_Estados -> ID Cor Lista2
Cor -> ( INT_CONST ,  INT_CONST,  INT_CONST ) |  cadeia_vazia
Grupos ->  GROUP  ID { Lista_de_Grupos }
Lista_de_Grupos ->  ID  Lista2
Lista2 -> ,  ID  Cor  Lista2  |   cadeia_vazia

Regras -> RULES { Regra RegraOpc }
RegraOpc -> Op  Regra RegraOpc | cadeia_vazia
Op-> AND | OR | XOR
Regra -> IF  ID  Expr  THEN  ID
Expr -> Op  ID Oprel INT_CONST  Expr  |  cadeia_vazia
Oprel -> > | >= | < | <= | != | ==

WIDTH 50 HEIGHT 50     //Cria o grid com o tamanho definido
STATE  { Alive (100, 50, 0), Dead(0,150,100)  }  //Criar um state e adicionar no final do vetor de estados
RULES {   IF  Alive AND Alive == 2 THEN Alive      //Cria uma regra e adiciona no final do vetor de regras
 IF  Alive AND Alive < 2 THEN Dead
}
